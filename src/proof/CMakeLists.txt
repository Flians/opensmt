PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}/PGBuild.C"
PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}/PGCheck.C"
PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}/PGHelp.C"

PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}/PGTransformationAlgorithms.C"
)
